            import java.util.*;

            public class p1026 {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    int k = scanner.nextInt();
                    int N = scanner.nextInt();
                    scanner.close();
                    List pow = new ArrayList<>();
                    int power = 1;
                    for (int i = 0; i <= 15; i++) {
                        pow.add(power);
                        power *= k;
                    }
                    // 使用集合来存储所有可能的和值
                    Set sums = new HashSet<>();
                    int m = pow.size();
                    // 生成所有可能的组合和
                    for (int mask = 0; mask < (1 << m); mask++) {
                        int sumss = 0;
                        for (int j = 0; j < m; j++) {
                            if ((mask & (1 << j)) != 0) {
                                sumss += pow.get(j);
                            }
                        }
                        sums.add(sumss);
                    }
                    List sortedSums = new ArrayList<>(sums);
                    Collections.sort(sortedSums);
                    System.out.println(sortedSums.get(N));
                }
            }
